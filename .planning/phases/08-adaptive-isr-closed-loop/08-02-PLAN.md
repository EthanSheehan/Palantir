       ```python
           if self._last_assessment is None:

           dispatches = []

               })
               nearest.tasking_source = "ISR_PRIORITY"
               nearest.mode_source = "AUTO"
               available.remove(nearest)
           return dispatches
       ```

       ```python
       def _generate_response_heuristic(
           self, detection: Detection, available_assets: List[SensorAsset]
       ) -> str:
           """Score assets by proximity and sensor match. No LLM needed."""
           import math
           orders = []
           det_lat = detection.coordinates.lat if hasattr(detection, 'coordinates') and detection.coordinates else 0.0
           det_lon = detection.coordinates.lon if hasattr(detection, 'coordinates') and detection.coordinates else 0.0
           for asset in available_assets:
               if asset.status != SensorStatusEnum.AVAILABLE:
                   continue
               a_lat = asset.location.lat if hasattr(asset, 'location') and asset.location else 0.0
               a_lon = asset.location.lon if hasattr(asset, 'location') and asset.location else 0.0
               dist = math.hypot(a_lon - det_lon, a_lat - det_lat)
               orders.append((dist, asset))
           orders.sort(key=lambda x: x[0])
           tasking_orders = []
           for dist, asset in orders[:2]:
               tasking_orders.append({
                   "asset_id": asset.asset_id if hasattr(asset, 'asset_id') else str(asset),
                   "collection_type": asset.capabilities[0] if hasattr(asset, 'capabilities') and asset.capabilities else "EO_IR",
                   "priority": "HIGH" if dist < 0.5 else "MEDIUM",
                   "reasoning": f"Nearest available asset at distance {dist:.3f} deg",
               })
           output = TaskingManagerOutput(
               tasking_orders=tasking_orders,
               confidence_gap=round(self.confidence_threshold - detection.confidence, 4),
               reasoning=f"Heuristic: assigned {len(tasking_orders)} nearest assets for secondary verification.",
           )
           return output.model_dump_json()
       ```
