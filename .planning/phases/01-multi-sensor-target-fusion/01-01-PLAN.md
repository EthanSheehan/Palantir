def fuse_detections(contributions: list[SensorContribution]) -> FusedDetection:
  1. Early return for empty list (all zeros/empty tuples)
  2. Group by sensor_type -> take max confidence per type (dict comprehension, single pass)
  3. Complementary fusion: complement = 1.0; for ci in per_type.values(): complement *= (1.0 - ci); fused = max(0.0, min(1.0, 1.0 - complement))
  4. Build sorted tuples for sensor_types and contributing_uav_ids
  5. Return FusedDetection
